SELECT nome, cognome
FROM clienti 
JOIN clienti_ordini co ON clienti.id = clienti_ordini.cliente_id
JOIN ordini o ON clienti_ordini.ordine_id = ordini.id
JOIN ordini_pizze ON ordini.id = ordini_pizze.ordine_id
JOIN pizze ON ordini_pizze.pizza_nome = pizze.nome
WHERE ordini.data = dataSpecifica
AND pizze.nome = pizzaSpecifica;

SELECT nome, cognome
FROM clienti 
JOIN clienti_ordini co ON clienti.id = clienti_ordini.cliente_id
JOIN ordini o ON clienti_ordini.ordine_id = ordini.id
JOIN ordini_pizze ON ordini.id = ordini_pizze.ordine_id
JOIN pizze ON ordini_pizze.pizza_nome = pizze.nome
JOIN pizze_ingredienti ON pizze.nome = pizze_ingredienti.pizza_nome
JOIN ingredienti ON pizze_ingredienti.ingrediente_nome = ingredienti.nome
WHERE ordini.data = dataSpecifica
AND pizze.nome = pizzaSpecifica
AND ordini.pagamento = 'cartaDiCredito'
AND ingredienti.nome = ingredienteSpecifica;


SELECT clienti.nome, clienti.cognome, COUNT(ordini.id) AS NumeroOrdini
FROM clienti 
JOIN ordini o ON clienti.id = ordini.cliente_id
GROUP BY cleinti.id, clienti.nome, clienti.cognome
ORDER BY NumeroOrdini DESC
LIMIT 1;
